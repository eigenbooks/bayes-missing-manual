import numpy as np
from myst_nb import glue

def ROM(n):
    exact = np.log10(n)
    rounded = np.floor(exact)
    print(f'Exact: {exact}')
    print(f'Rounded: {rounded}')
    return exact, rounded

log124, log124_floor = ROM(124)
glue("log124", log124, display=False)
glue("log124_floor", log124_floor, display=False)

Rounded: 2.0
